.section .text
.globl lab02d 

lab02d:
    ADD X0, X0, #1000 		    // add 1000 to the value in X9
    ADRP X10, myvalue               //store first 20 bits in X10. //X3
    ADD X10, X10, :lo12:myvalue     //store lower 12 bits in X10
    STUR X0, [X10, 0] 		    //Save the updated value from X9 to X10
    MOV X0, X10 		    //return address of myvalue
    BR X30

.section .data

myvalue:
    .quad 0x0
